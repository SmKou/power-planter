const feed = changeState("soil")(1);

const feedPlant = fn => plant => ({
    ...plant,
    feed: function () { return fn(this) }
})

const red_plant = name => {
    const state = { name }
    return {
        ...state,
        ...feedPlant(feed)(state)
    }
};

const pink_plant = stateControl(feed);
const maroon_plant = stateControl(feedPlant(feed));

module.exports = {
    pink_plant,
    red_plant,
    maroon_plant,
    feedPlant,
    feed
}
